    60→// --- Auto-configure missing env vars (write to .env silently) ---
    61→// NOTE: This function is kept for non-secret env vars only (e.g., APP_URL).
    62→// It MUST NOT be used for secrets (JWT_SECRET, ADMIN_PASSWORD, etc.) —

    65→function autoSetEnv(key: string, value: string) {
    66→  process.env[key] = value;
    67→  try {
    68→    const envContent = fs.existsSync(envPath) ? fs.readFileSync(envPath, 'utf-8') : '';
    69→    const lines = envContent.split('\n');
    70→    const keyPattern = new RegExp(`^${key}=`);
    71→    const existingIndex = lines.findIndex(l => keyPattern.test(l));
    72→    if (existingIndex >= 0) {
    73→      lines[existingIndex] = `${key}=${value}`;
    74→    } else {
    75→      lines.push(`${key}=${value}`);
    76→    }
    77→    fs.writeFileSync(envPath, lines.join('\n'), 'utf-8');
    78→    // Also save to persistent .env if different from main .env
    79→    const persistentEnvPath = path.resolve(PERSISTENT_DIR, '.env');
    80→    if (persistentEnvPath !== envPath) {
    81→      try {
    82→        const pEnvContent = fs.existsSync(persistentEnvPath) ? fs.readFileSync(persistentEnvPath, 'utf-8') : '';
    83→        const pLines = pEnvContent.split('\n');
    84→        const pExistingIndex = pLines.findIndex(l => keyPattern.test(l));
    85→        if (pExistingIndex >= 0) {
    86→          pLines[pExistingIndex] = `${key}=${value}`;
    87→        } else {
    88→          pLines.push(`${key}=${value}`);
    89→        }
    90→        fs.writeFileSync(persistentEnvPath, pLines.join('\n'), 'utf-8');
    91→      } catch { /* ignore */ }
    92→    }
    93→  } catch { /* ignore write errors */ }
    94→}

   522→  // ===========================================
   523→  // 7d. Device registration for push notifications
   524→  // Auth is required: an unauthenticated caller must not be able to
   525→  // supply an arbitrary token and DELETE/UPDATE rows belonging to other
   526→  // users. Guest (anonymous) device registration is intentionally removed
   527→  // — only logged-in users receive push notifications anyway.
   528→  // ===========================================
   529→  app.post('/api/notifications/register-device', async (req, res) => {
   530→    try {
   531→      const { token, platform } = req.body;
   532→      if (!token || !platform) {
   533→        res.status(400).json({ error: 'token and platform required' });
   534→        return;
   535→      }
   536→      const authHeader = req.headers.authorization;
   537→      if (!authHeader?.startsWith('Bearer ')) {
   538→        res.status(401).json({ error: 'Unauthorized' });
   539→        return;
   540→      }
   541→      const { verifyToken } = await import('./middleware/auth.js');
   542→      const payload = verifyToken(authHeader.split(' ')[1]);
   543→      if (!payload) {
   544→        res.status(401).json({ error: 'Invalid or expired token' });
   545→        return;
   546→      }
   547→      const userId: string = payload.userId || payload.sub || '';
   548→      if (!userId) {
   549→        res.status(401).json({ error: 'Unauthorized' });
   550→        return;
   551→      }
   552→      // Save or update device token in database
   553→      try {
   554→        const db = (await import('./database/index.js')).default;
   555→        // Remove token from any other user (a token can only belong to one user)
   556→        db.prepare('DELETE FROM devices WHERE token = ? AND user_id != ?').run(token, userId);
   557→        // Upsert device token
   558→        const existing = db.prepare('SELECT id FROM devices WHERE token = ?').get(token) as any;
   559→        if (existing) {
   560→          db.prepare('UPDATE devices SET user_id = ?, platform = ?, updated_at = datetime(\'now\') WHERE token = ?').run(userId, platform, token);
   561→        } else {
   562→          db.prepare('INSERT INTO devices (user_id, token, platform) VALUES (?, ?, ?)').run(userId, token, platform);
   563→        }
   564→      } catch (dbErr: any) {
   565→        console.warn('[FCM] Failed to save device token:', dbErr.message);
   566→      }
   567→      console.log(`[FCM] Device registered: platform=${platform}, user=${userId}`);
   568→      res.json({ success: true, registered: true });
   569→    } catch (err: any) {
   570→      console.error('[FCM] register-device error:', err.message);
   571→      res.status(500).json({ error: 'Failed to register device' });
   572→    }
   573→  });

   605→      if (!title || !body) {

   608→      }
   609→
   610→      // Broadcast to all users

   614→        const { sendPushToUsers } = await import('./services/pushNotifications.js');
   615→        const result = await sendPushToUsers(allUsers.map(u => u.id), title, body, data);
   616→        res.json({ success: true, broadcast: true, totalUsers: allUsers.length, ...result });
   617→        return;
   618→      }
   619→
   620→      // Send to specific user IDs list
   621→      if (targetUserIds && Array.isArray(targetUserIds) && targetUserIds.length > 0) {
   622→        const { sendPushToUsers } = await import('./services/pushNotifications.js');
   623→        const result = await sendPushToUsers(targetUserIds, title, body, data);
   624→        res.json({ success: true, ...result });
   625→        return;
   626→      }
   627→
   628→      // Send to a topic
   629→      if (topic) {
   630→        const { sendPushToTopic } = await import('./services/pushNotifications.js');
   631→        const result = await sendPushToTopic(topic, title, body, data);
   632→        res.json({ success: true, ...result });
   633→        return;
   634→      }
   635→
   636→      // Send to a single user
   637→      if (!targetUserId) {
   638→        res.status(400).json({ error: 'userId, userIds, or topic required' });
   639→        return;
   640→      }
   641→      const { sendPushToUser } = await import('./services/pushNotifications.js');
   642→      const result = await sendPushToUser(targetUserId, title, body, data);
   643→      res.json({ success: true, ...result });

  1042→  app.post('/api/livestream/gift', async (req, res) => {
  1043→    try {
  1044→      const db = (await import('./database/index.js')).default;
  1045→      const authHeader = req.headers.authorization;
  1046→      if (!authHeader?.startsWith('Bearer ')) { res.status(401).json({ error: 'Unauthorized' }); return; }
  1047→      const { verifyToken } = await import('./middleware/auth.js');
  1048→      const payload = verifyToken(authHeader.split(' ')[1]);
  1049→      if (!payload) { res.status(401).json({ error: 'Invalid token' }); return; }
  1050→      const userId = payload.userId || payload.sub;
  1051→      const { streamId, receiverId, giftType, message } = req.body;
  1052→      if (!streamId || !receiverId || !giftType) { res.status(400).json({ error: 'streamId, receiverId, giftType required' }); return; }
  1053→
  1054→      const gift = GIFT_TYPES.find(g => g.id === giftType);
  1055→      if (!gift) { res.status(400).json({ error: 'Invalid gift type' }); return; }
  1056→
  1057→      // Atomic balance check + deduction + receiver credit + gift record + tx records.
  1058→      // All five operations must either succeed together or fail together so
  1059→      // we never lose or duplicate money on a partial failure.
  1060→      const sender = db.prepare('SELECT wallet_balance, name FROM users WHERE id = ?').get(userId) as any;
  1061→      if (!sender || sender.wallet_balance < gift.amount) { res.status(400).json({ error: 'رصيد غير كافي' }); return; }
  1062→
  1063→      const receiverAmount = gift.amount * 0.9;
  1064→      const giftId = crypto.randomBytes(16).toString('hex');
  1065→      const txId1 = crypto.randomBytes(16).toString('hex');
  1066→      const txId2 = crypto.randomBytes(16).toString('hex');
  1067→      const notifId = crypto.randomBytes(16).toString('hex');
  1068→
  1069→      db.transaction(() => {
  1070→        // Conditional UPDATE: only succeeds if balance is still ≥ amount.
  1071→        // Closes the race where two concurrent gifts both pass the outer check.
  1072→        const deductResult = db.prepare(
  1073→          'UPDATE users SET wallet_balance = wallet_balance - ? WHERE id = ? AND wallet_balance >= ?'
  1074→        ).run(gift.amount, userId, gift.amount);
  1075→        if (deductResult.changes === 0) {
  1076→          throw new Error('INSUFFICIENT_BALANCE');
  1077→        }
  1078→        db.prepare('UPDATE users SET wallet_balance = wallet_balance + ? WHERE id = ?').run(receiverAmount, receiverId);
  1079→
  1080→        db.prepare('INSERT INTO stream_gifts (id, stream_id, sender_id, receiver_id, gift_type, gift_name, amount, message) VALUES (?, ?, ?, ?, ?, ?, ?, ?)').run(
  1081→          giftId, streamId, userId, receiverId, giftType, gift.name, gift.amount, message || ''
  1082→        );
  1083→
  1084→        db.prepare('INSERT INTO transactions (id, user_id, type, amount, method, status) VALUES (?, ?, ?, ?, ?, ?)').run(
  1085→          txId1, userId, 'gift_sent', gift.amount, 'wallet', 'completed'
  1086→        );
  1087→        db.prepare('INSERT INTO transactions (id, user_id, type, amount, method, status) VALUES (?, ?, ?, ?, ?, ?)').run(
  1088→          txId2, receiverId, 'gift_received', receiverAmount, 'wallet', 'completed'
  1089→        );
  1090→
  1091→        db.prepare('INSERT INTO notifications (id, user_id, type, message, link) VALUES (?, ?, ?, ?, ?)').run(
  1092→          notifId, receiverId, 'payment', `استلمت ${gift.name} من ${sender.name}`, '/wallet'
  1093→        );
  1094→      })();
  1095→
  1096→      // Notify receiver via WebSocket (outside the transaction).
  1097→      // Use sendToUser with a raw object (NOT pre-stringified) so the
  1098→      // WS layer's internal JSON.stringify produces the correct shape.
  1099→      try {
  1100→        const { wsManager } = await import('./websocket/index.js');

  1102→          type: 'livestream:gift',
  1103→          data: { gift: { id: giftId, giftType, giftName: gift.name, amount: gift.amount, senderName: sender.name, message } }
  1104→        });
